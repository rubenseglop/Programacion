package multiplicarmatrices;

/**
 *
 * @author ruben
 */
public class MultiplicarMatrices {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) throws InterruptedException {
        // TODO code application logic here
        int[][] resultado;
        int x1 = 3000;
        int y1 = 3000;
        int x2 = 3000;
        int y2 = 3000;
        Matriz mimatriz = new Matriz(x1, y1, x2, y2);
        mimatriz.setValoresAuto(x1, y1, x2, y2);
        long startTime = 0;

// Mostrar matrices
        startTime = System.currentTimeMillis();
        mimatriz.mostrarMatriz();

// multiplicar arrays
        resultado = new int[x1][y2];
        FilasHebras hebra1 = new FilasHebras(resultado, mimatriz, 0,750);
        FilasHebras hebra2 = new FilasHebras(resultado, mimatriz, 751, 1500);
        FilasHebras hebra3 = new FilasHebras(resultado, mimatriz, 1501, 2250);
        FilasHebras hebra4 = new FilasHebras(resultado, mimatriz, 2251, 3000);

        hebra1.start();
        hebra2.start();
        hebra3.start();
        hebra4.start();
        hebra1.join();
        hebra2.join();
        hebra3.join();
        hebra4.join();

        System.out.println("Matriz resultado");
        for (int i = 0; i < x1; i++) {
            System.out.print("[ ");
            for (int j = 0; j < y2; j++) {
                System.out.print(resultado[i][j] + " ");
            }
            System.out.print("]\n");
        }
        long endTime = System.currentTimeMillis() - startTime;
        System.out.println("\nTiempo en milisegundos " + endTime);
    }
}
