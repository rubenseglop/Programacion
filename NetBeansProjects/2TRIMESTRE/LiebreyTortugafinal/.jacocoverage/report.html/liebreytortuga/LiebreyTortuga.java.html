package liebreytortuga;

import java.util.Scanner;

/**
 *
 * @author ruben
 */
public class LiebreyTortuga {

    /**
     * @param args the command line arguments
     */
    public static void main(String[] args) {

        //Creo el reloj que sincronizará a los competidores
        Carrera carrera = new Carrera();

        //Creando a los competidores
        Liebre liebre1 = new Liebre(carrera, 1);
        Tortuga tortuga1 = new Tortuga(carrera, 2);
        Tortuga tortuga2 = new Tortuga(carrera, 3);
        Liebre liebre2 = new Liebre(carrera, 4);

        // Añadiendo competidores a la carrera
        liebre1.start();
        tortuga1.start();
        tortuga2.start();
        liebre2.start();

        String imprimir = "";
        for (int i = 0; i <= 70; i++) {
            imprimir = imprimir + "-";

        }
        int fincarrera = 0; // Determino cuando acabar con la carrera 
        while (fincarrera == 0) { // los if determinan a un sólo ganador y sale del bucle


            if (liebre1.isAlive() == false) {
                liebre2.finProceso();
                tortuga1.finProceso();
                tortuga2.finProceso();
                System.out.println(imprimir + "METAAAAA!!! para la Liebre 1");

                fincarrera++;
            }
            if (liebre2.isAlive() == false) {
                liebre1.finProceso();
                tortuga1.finProceso();
                tortuga2.finProceso();

                System.out.println(imprimir + "METAAAAA!!! para la Liebre 2");

                fincarrera++;
            }
            if (tortuga1.isAlive() == false) {

                liebre1.finProceso();
                liebre2.finProceso();
                tortuga2.finProceso();
                System.out.println(imprimir + "METAAAAA!!! para la Tortuga 1");

                fincarrera++;
            }
            if (tortuga2.isAlive() == false) {

                liebre1.finProceso();
                liebre2.finProceso();
                tortuga1.finProceso();
                System.out.print(imprimir + "METAAAAA!!! para la Tortuga 2");

                fincarrera++;
            }

        }

        System.out.println(" Fin carrera");
    }
}
